class Solution {
public:
    int maximizeSweetness(vector<int>& sweetness, int k) {
        int sumSw=0;
        for(int i=0;i<sweetness.size();i++){
            sumSw+=sweetness[i];
        }
        int avg=sumSw/(k+1);
        int l=0,r=avg;
        while(l<=r){
            int mid=(r-l)/2+l;
            if(isFeasible(mid, k,sweetness))
            l=mid+1;
            else r=mid-1;
        }
        return r;
    }
    bool isFeasible(int minDiv,int k,vector<int> sweetness){
        int count=0,sum=0;
        for(int x:sweetness){
            sum+=x;
            if(sum>=minDiv){
                sum=0;
                count++;
            }
        }
        return count>=k+1;
    }
};

class Solution:
    def maximizeSweetness(self, sweetness: List[int], K: int) -> int:
        l,r = min(sweetness)-1,sum(sweetness)+1
        while l+1!=r:
            mid = (l+r)//2
            k = 0
            c = 0
            for sw in sweetness:
                c += sw
                if c>=mid:
                    c = 0
                    k += 1
            if k>=K+1:
                l = mid
            else:
                r = mid
        return l
